#replace funny characters with %xx hex for urls
sub escape
{
    ($_)=@_;
    s/([^a-zA-Z0-9_\-.])/uc sprintf("%%%02x",ord($1))/eg;
    $_;
}

#replace + with space and %xx with that ASCII character
sub unescape {
    ($_)=@_;
    tr/+/ /;
    s/%(..)/pack("c",hex($1))/ge;
    $_;
}
